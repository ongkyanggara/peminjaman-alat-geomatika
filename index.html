function doGet() { return ContentService .createTextOutput(JSON.stringify(getData())) .setMimeType(ContentService.MimeType.JSON); } function getData() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); const data = sheet.getDataRange().getValues(); const headers = data.shift(); return data.map(row => { let obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } function doPost(e) { const data = JSON.parse(e.postData.contents); if (data.action === "create") { return createData(data); } if (data.action === "updateStatus") { return updateStatus(data); } if (data.action === "delete") { return deleteData(data); } } function createData(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); let fileUrl = ""; // =============================== // HANDLE FILE (BASE64 → DRIVE) // =============================== if (data.file && data.fileName) { const blob = Utilities.newBlob( Utilities.base64Decode(data.file), data.mimeType || "application/pdf", data.fileName ); const folder = DriveApp.getFolderById("PASTE_FOLDER_ID_DI_SINI"); const file = folder.createFile(blob); fileUrl = file.getUrl(); } sheet.appendRow([ data.id, data.borrowerName, data.nim, data.jenis, data.borrowDate, data.borrowTime, data.returnDateEst, data.returnTime, "active", fileUrl, JSON.stringify(data.items) ]); return ContentService.createTextOutput( JSON.stringify({ status: "success" }) ).setMimeType(ContentService.MimeType.JSON); } function updateStatus(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sheet.getRange(i + 1, 9).setValue(data.status); // kolom status break; } } return ContentService.createTextOutput("updated"); } function deleteData(data) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === data.id) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput("deleted"); }
